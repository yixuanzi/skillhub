```python
import asyncio
import json
import subprocess
from typing import Any, Optional
from sqlalchemy.orm import Session

from backend.models.resource import Resource
from backend.schemas.resource import MCPConfig, MCPServerType
from backend.services.token_service import TokenService
from backend.core.exceptions import ValidationException, ExternalServiceException

class MCPService:
    """Service for calling MCP (Model Context Protocol) servers"""

    @staticmethod
    def parse_mcp_config(ext: dict) -> MCPConfig:
        """Parse and validate MCP config from resource ext"""
        if not ext:
            raise ValidationException("MCP resources must have ext configuration")
        return MCPConfig(**ext)

    @staticmethod
    async def call_mcp_resource(
        db: Session,
        resource_id: str,
        method: str,
        params: dict[str, Any]
    ) -> dict[str, Any]:
        """
        Call an MCP server resource

        Args:
            db: Database session
            resource_id: Resource ID
            method: JSON-RPC method name
            params: JSON-RPC parameters

        Returns:
            JSON-RPC response result
        """
        resource = db.query(Resource).filter(Resource.id == resource_id).first()
        if not resource:
            raise ValidationException("Resource not found")

        if resource.type != "mcp":
            raise ValidationException("Resource is not an MCP resource")

        config = MCPService.parse_mcp_config(resource.ext)

        # Replace token placeholders in env vars
        if config.env:
            config.env = MCPService._replace_tokens(db, config.env)

        if config.transport == MCPServerType.STDIO:
            return await MCPService._call_stdio(config, method, params)
        elif config.transport == MCPServerType.SSE:
            return await MCPService._call_sse(config, method, params)
        elif config.transport == MCPServerType.WS:
            return await MCPService._call_ws(config, method, params)
        else:
            raise ValidationException(f"Unsupported transport: {config.transport}")

    @staticmethod
    def _replace_tokens(db: Session, env: dict[str, str]) -> dict[str, str]:
        """Replace ${token:name} placeholders with actual token values"""
        token_service = TokenService()
        result = {}

        for key, value in env.items():
            if isinstance(value, str) and "${token:" in value:
                # Extract token name: ${token:api_name} -> api_name
                import re
                match = re.search(r'\$\{token:([^}]+)\}', value)
                if match:
                    token_name = match.group(1)
                    # Get actual token value from token service
                    token = token_service.get_token_by_name(db, token_name)
                    if token:
                        value = value.replace(f'${{token:{token_name}}}', token.value)
                    else:
                        raise ValidationException(f"Token not found: {token_name}")
            result[key] = value

        return result

    @staticmethod
    async def _call_stdio(config: MCPConfig, method: str, params: dict) -> dict:
        """
        Execute MCP server via STDIO transport

        Spawns subprocess and communicates via JSON-RPC over stdin/stdout
        """
        # Build command
        cmd = [config.command]
        if config.args:
            cmd.extend(config.args)

        # Prepare JSON-RPC request
        request = {
            "jsonrpc": "2.0",
            "id": 1,
            "method": method,
            "params": params
        }

        try:
            # Start process
            process = await asyncio.create_subprocess_exec(
                *cmd,
                stdin=asyncio.subprocess.PIPE,
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE,
                env={**dict(config.env)} if config.env else None
            )

            # Send request
            request_json = json.dumps(request) + "\n"
            process.stdin.write(request_json.encode())
            await process.stdin.drain()

            # Read response (with timeout)
            try:
                stdout, stderr = await asyncio.wait_for(
                    process.communicate(),
                    timeout=config.timeout / 1000
                )
            except asyncio.TimeoutError:
                process.kill()
                raise ExternalServiceException("MCP server timeout")

            if process.returncode != 0:
                error_msg = stderr.decode() if stderr else "Unknown error"
                raise ExternalServiceException(f"MCP server error: {error_msg}")

            # Parse response
            response = json.loads(stdout.decode().strip())

            if "error" in response:
                raise ExternalServiceException(f"MCP error: {response['error']}")

            return response.get("result", {})

        except FileNotFoundError:
            raise ExternalServiceException(f"Command not found: {config.command}")
        except json.JSONDecodeError:
            raise ExternalServiceException("Invalid JSON response from MCP server")

    @staticmethod
    async def _call_sse(config: MCPConfig, method: str, params: dict) -> dict:
        """
        Execute MCP server via Server-Sent Events transport

        Connects to SSE endpoint and sends JSON-RPC request
        """
        import httpx

        request = {
            "jsonrpc": "2.0",
            "id": 1,
            "method": method,
            "params": params
        }

        try:
            async with httpx.AsyncClient(timeout=config.timeout / 1000) as client:
                response = await client.post(
                    config.endpoint,
                    json=request,
                    headers={"Content-Type": "application/json"}
                )
                response.raise_for_status()

                result = response.json()

                if "error" in result:
                    raise ExternalServiceException(f"MCP error: {result['error']}")

                return result.get("result", {})

        except httpx.HTTPError as e:
            raise ExternalServiceException(f"SSE connection error: {e}")

    @staticmethod
    async def _call_ws(config: MCPConfig, method: str, params: dict) -> dict:
        """
        Execute MCP server via WebSocket transport

        Connects to WebSocket endpoint and sends JSON-RPC request
        """
        import websockets
        import asyncio

        request = {
            "jsonrpc": "2.0",
            "id": 1,
            "method": method,
            "params": params
        }

        try:
            async with websockets.connect(
                config.endpoint,
                close_timeout=config.timeout / 1000
            ) as ws:
                await ws.send(json.dumps(request))

                response_str = await asyncio.wait_for(
                    ws.recv(),
                    timeout=config.timeout / 1000
                )

                response = json.loads(response_str)

                if "error" in response:
                    raise ExternalServiceException(f"MCP error: {response['error']}")

                return response.get("result", {})

        except asyncio.TimeoutError:
            raise ExternalServiceException("WebSocket timeout")
        except Exception as e:
            raise ExternalServiceException(f"WebSocket error: {e}")
```

from typing import List, Optional
from sqlalchemy.orm import Session

from backend.models.resource import Resource, ViewScope
from backend.models.user import User
from backend.schemas.resource import ResourceCreate, ResourceUpdate
from backend.core.exceptions import ValidationException, NotFoundException

class ResourceService:
    @staticmethod
    def create(db: Session, data: ResourceCreate, user: User) -> Resource:
        """Create a new resource"""
        # Validate unique name
        existing = db.query(Resource).filter(Resource.name == data.name).first()
        if existing:
            raise ValidationException("Resource name already exists")

        # Private resources must have owner
        if data.view_scope == ViewScope.PRIVATE and not data.owner_id:
            data.owner_id = user.id

        # Validate MCP config
        if data.type == ResourceType.MCP:
            if not data.mcp_config:
                raise ValidationException("MCP resources require mcp_config")
            # Store MCP config in ext
            data.ext = data.mcp_config.model_dump()

        resource = Resource(
            name=data.name,
            desc=data.desc,
            type=data.type.value,
            url=data.url,
            _ext=json.dumps(data.ext) if data.ext else None,
            view_scope=data.view_scope.value,
            owner_id=data.owner_id,
            api_description=data.api_description
        )

        db.add(resource)
        db.commit()
        db.refresh(resource)

        return resource

    @staticmethod
    def list_accessible(db: Session, user: User, skip: int = 0, limit: int = 100) -> List[Resource]:
        """
        List resources accessible to user respecting view_scope and ACL

        - Public resources: visible to all authenticated users
        - Private resources: only visible to owner or users with ACL permission
        """
        from backend.models.acl_rule import ACLRule
        from backend.models.role_binding import RoleBinding
        from backend.models.user_role import UserRole

        # Public resources
        public_query = db.query(Resource).filter(
            Resource.view_scope == ViewScope.PUBLIC.value
        )

        # Private resources owned by user
        owned_query = db.query(Resource).filter(
            Resource.view_scope == ViewScope.PRIVATE.value,
            Resource.owner_id == user.id
        )

        # Private resources with ACL access
        acl_query = db.query(Resource).join(ACLRule).join(RoleBinding).join(UserRole).filter(
            Resource.view_scope == ViewScope.PRIVATE.value,
            UserRole.user_id == user.id
        )

        # Combine and deduplicate
        all_resources = list(set(
            public_query.all() + owned_query.all() + acl_query.all()
        ))

        return all_resources[skip:skip + limit]

    @staticmethod
    def get_accessible(db: Session, resource_id: str, user: User) -> Resource:
        """Get resource if user has access (view_scope or ACL)"""
        resource = db.query(Resource).filter(Resource.id == resource_id).first()
        if not resource:
            raise NotFoundException("Resource not found")

        # Public: always accessible
        if resource.view_scope == ViewScope.PUBLIC.value:
            return resource

        # Private: check ownership or ACL
        if resource.view_scope == ViewScope.PRIVATE.value:
            if resource.owner_id == user.id:
                return resource

            # Check ACL permission
            from backend.services.acl_resource_service import ACLResourceService
            has_access = ACLResourceService.check_permission(
                db, resource_id, user.id, "read"
            )
            if has_access:
                return resource

            raise ValidationException("Private resource - no access")

        raise NotFoundException("Resource not found")

    @staticmethod
    def update(db: Session, resource_id: str, data: ResourceUpdate, user: User) -> Resource:
        """Update resource"""
        resource = ResourceService.get_accessible(db, resource_id, user)

        update_data = data.model_dump(exclude_unset=True)
        for field, value in update_data.items():
            setattr(resource, field, value)

        db.commit()
        db.refresh(resource)

        return resource

    @staticmethod
    def delete(db: Session, resource_id: str, user: User) -> Resource:
        """Delete resource"""
        resource = ResourceService.get_accessible(db, resource_id, user)

        # Only owner or admin can delete
        if resource.owner_id != user.id:
            # Check if user is admin
            if not any(role.name == "admin" for role in user.roles):
                raise ValidationException("Only resource owner or admin can delete")

        db.delete(resource)
        db.commit()

        return resource

from sqlalchemy.orm import Session
from typing import Optional
from backend.models.resource import Resource
from backend.models.acl_rule import ACLRule
from backend.models.user import User
from backend.core.exceptions import ValidationException

class ACLResourceService:
    @staticmethod
    def check_permission(
        db: Session,
        resource_id: str,
        user_id: str,
        required_permission: str
    ) -> bool:
        """
        Check if user has permission to access resource

        Considers:
        1. Public resources - always visible
        2. Private resources - owner or ACL permission required
        3. ACL rules - role-based access control
        """
        resource = db.query(Resource).filter(Resource.id == resource_id).first()
        if not resource:
            return False

        # Public resources: always accessible for visibility
        if resource.view_scope == "public":
            # But still check ACL for actual operations
            pass
        else:
            # Private resources: check ownership first
            if resource.owner_id == user_id:
                return True

        # Check ACL rules
        acl_rule = db.query(ACLRule).filter(
            ACLRule.resource_id == resource_id
        ).first()

        if not acl_rule:
            # No ACL rule means no access for private resources
            return resource.view_scope == "public"

        # Any mode - public access
        if acl_rule.access_mode == "any":
            return True

        # RBAC mode - check role bindings
        if acl_rule.access_mode == "rbac":
            from backend.models.role_binding import RoleBinding
            from backend.models.user_role import UserRole

            user_roles = db.query(UserRole).filter(
                UserRole.user_id == user_id
            ).all()

            user_role_ids = [ur.role_id for ur in user_roles]

            for binding in acl_rule.role_bindings:
                if binding.role_id in user_role_ids:
                    if required_permission in binding.permissions:
                        return True

        return False

    @staticmethod
    def get_accessibility_info(
        db: Session,
        resource_id: str,
        user_id: str
    ) -> dict:
        """
        Get accessibility information for a resource

        Returns:
            {
                "is_accessible": bool,
                "access_reason": str,
                "can_read": bool,
                "can_write": bool,
                "can_execute": bool
            }
        """
        resource = db.query(Resource).filter(Resource.id == resource_id).first()
        if not resource:
            return {"is_accessible": False, "access_reason": "Resource not found"}

        # Check visibility
        is_visible = False
        if resource.view_scope == "public":
            is_visible = True
        elif resource.owner_id == user_id:
            is_visible = True
        elif ACLResourceService.check_permission(db, resource_id, user_id, "read"):
            is_visible = True

        if not is_visible:
            return {
                "is_accessible": False,
                "access_reason": "Private resource - not visible"
            }

        # Check permissions
        return {
            "is_accessible": True,
            "access_reason": "Resource is accessible",
            "can_read": ACLResourceService.check_permission(db, resource_id, user_id, "read"),
            "can_write": ACLResourceService.check_permission(db, resource_id, user_id, "write"),
            "can_execute": ACLResourceService.check_permission(db, resource_id, user_id, "execute")
        }

from sqlalchemy.orm import Session
from typing import List, Optional

from backend.models.user import User
from backend.models.role import Role
from backend.models.user_role import UserRole
from backend.schemas.user_management import RoleCreate, RoleAssignRequest
from backend.core.exceptions import ValidationException, NotFoundException

class UserManagementService:
    @staticmethod
    def list_users(
        db: Session,
        skip: int = 0,
        limit: int = 100,
        search: Optional[str] = None,
        role_id: Optional[str] = None
    ) -> tuple[List[User], int]:
        """List users with filters (admin only)"""
        query = db.query(User)

        if search:
            query = query.filter(
                (User.username.ilike(f"%{search}%")) |
                (User.email.ilike(f"%{search}%"))
            )

        if role_id:
            query = query.join(UserRole).filter(UserRole.role_id == role_id)

        total = query.count()
        users = query.offset(skip).limit(limit).all()

        return users, total

    @staticmethod
    def get_user_details(db: Session, user_id: str) -> User:
        """Get user details with roles"""
        user = db.query(User).filter(User.id == user_id).first()
        if not user:
            raise NotFoundException("User not found")
        return user

    @staticmethod
    def assign_roles(
        db: Session,
        user_id: str,
        role_ids: List[str],
        admin_user_id: str
    ) -> User:
        """Assign roles to user (admin only)"""
        user = UserManagementService.get_user_details(db, user_id)

        # Validate roles exist
        valid_roles = db.query(Role).filter(Role.id.in_(role_ids)).all()
        if len(valid_roles) != len(role_ids):
            raise ValidationException("One or more roles not found")

        # Remove existing roles
        db.query(UserRole).filter(UserRole.user_id == user_id).delete()

        # Add new roles
        for role_id in role_ids:
            user_role = UserRole(user_id=user_id, role_id=role_id)
            db.add(user_role)

        db.commit()
        db.refresh(user)

        return user

    @staticmethod
    def list_roles(db: Session) -> List[Role]:
        """List all available roles"""
        return db.query(Role).all()

    @staticmethod
    def create_role(db: Session, data: RoleCreate) -> Role:
        """Create a new role (admin only)"""
        # Check if role name already exists
        existing = db.query(Role).filter(Role.name == data.name).first()
        if existing:
            raise ValidationException("Role name already exists")

        role = Role(name=data.name, description=data.description)
        db.add(role)
        db.commit()
        db.refresh(role)

        return role

    @staticmethod
    def deactivate_user(
        db: Session,
        user_id: str,
        admin_user_id: str
    ) -> User:
        """Deactivate a user (admin only)"""
        user = UserManagementService.get_user_details(db, user_id)

        # Prevent deactivating self
        if user_id == admin_user_id:
            raise ValidationException("Cannot deactivate your own account")

        user.is_active = False
        db.commit()
        db.refresh(user)

        return user

**Step 4: Create Role Manager Modal**

Create `frontend/src/pages/users/components/RoleManagerModal.tsx`:

```typescript
import React, { useState } from 'react';
import { Dialog, DialogContent, DialogDescription, DialogHeader, DialogTitle } from '@/components/ui/dialog';
import { Button } from '@/components/ui/button';
import { Label } from '@/components/ui/label';
import { Checkbox } from '@/components/ui/checkbox';
import { useRoles, useAssignRoles } from '@/hooks/use-user-management';

interface Props {
  user: User;
  open: boolean;
  onOpenChange: (open: boolean) => void;
}

export const RoleManagerModal: React.FC<Props> = ({ user, open, onOpenChange }) => {
  const { data: roles, isLoading } = useRoles();
  const assignRoles = useAssignRoles();

  const [selectedRoleIds, setSelectedRoleIds] = useState<string[]>(
    user.roles.map(r => r.id)
  );

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    assignRoles.mutate(
      { userId: user.id, roleIds: selectedRoleIds },
      {
        onSuccess: () => {
          onOpenChange(false);
        },
      }
    );
  };

  const toggleRole = (roleId: string) => {
    setSelectedRoleIds(prev =>
      prev.includes(roleId)
        ? prev.filter(id => id !== roleId)
        : [...prev, roleId]
    );
  };

  if (isLoading) {
    return null;
  }

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Manage Roles</DialogTitle>
          <DialogDescription>
            Assign roles to {user.username}
          </DialogDescription>
        </DialogHeader>

        <form onSubmit={handleSubmit} className="space-y-4">
          <div>
            <Label>Roles</Label>
            <div className="space-y-2 mt-2">
              {roles?.map((role) => (
                <div key={role.id} className="flex items-center space-x-2">
                  <Checkbox
                    id={role.id}
                    checked={selectedRoleIds.includes(role.id)}
                    onCheckedChange={() => toggleRole(role.id)}
                  />
                  <Label htmlFor={role.id} className="text-sm cursor-pointer">
                    {role.name}
                  </Label>
                  {role.description && (
                    <span className="text-xs text-muted-foreground">
                      - {role.description}
                    </span>
                  )}
                </div>
              ))}
            </div>
          </div>

          <div className="flex justify-end space-x-2">
            <Button
              type="button"
              variant="outline"
              onClick={() => onOpenChange(false)}
            >
              Cancel
            </Button>
            <Button type="submit" disabled={assignRoles.isPending}>
              {assignRoles.isPending ? 'Saving...' : 'Save Changes'}
            </Button>
          </div>
        </form>
      </DialogContent>
    </Dialog>
  );
};
```

MCP (Model Context Protocol) resources enable integration with MCP servers for LLM tool integration.

}
\`\`\`

Environment variables support token placeholders:

\`\`\`json
